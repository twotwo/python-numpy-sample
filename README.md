## setup for running

    # newer for poetry
    curl -sSL https://install.python-poetry.org | POETRY_VERSION=1.2.1 python3 -
    # install dependency
    poetry install
    poetry shell
    # install pre-commit and run checks
    pre-commit install
    pre-commit run -a
    # launch notebook
    jupyter notebook
